# C语言基础语法

本教程将帮助你系统地学习C语言的基础语法，包括变量、数据类型、运算符等内容。

## 1. 第一个C程序

int main() {
    printf("Hello, World!\n");
    return 0;
}
```

gcc hello.c -o hello

```

## 2. 变量和数据类型

### 基本数据类型

C语言的基本数据类型包括：

- **整型**: `int`, `short`, `long`, `long long`
- **字符型**: `char`
- **浮点型**: `float`, `double`
- **布尔型**: `_Bool` (C99标准)

```c
#include <stdio.h>

int main() {
    // 整数类型
    int age = 25;
    short score = 85;
    long big_number = 1000000L;
    
    // 字符类型
    char grade = 'A';
    
    // 浮点数类型
    float height = 175.5f;
    double pi = 3.14159265359;
    
    // 布尔类型（需要包含stdbool.h）
    // bool is_student = true;
    
    printf("年龄: %d\n", age);
    printf("分数: %hd\n", score);
    printf("大数: %ld\n", big_number);
    printf("等级: %c\n", grade);
    printf("身高: %.2f\n", height);
    printf("圆周率: %.10f\n", pi);
    
    return 0;
}
```

### 变量声明和初始化

```c
#include <stdio.h>

int main() {
    // 先声明后赋值
    int x;
    x = 10;
    
    // 声明并初始化
    int y = 20;
    
    // 多个变量声明
    int a = 1, b = 2, c = 3;
    
    printf("x = %d, y = %d\n", x, y);
    printf("a = %d, b = %d, c = %d\n", a, b, c);
    
    return 0;
}
```

### 常量

```c
#include <stdio.h>

#define PI 3.14159          // 宏定义常量
#define MAX_SIZE 100

int main() {
    // const关键字定义常量
    const int MIN_AGE = 18;
    const float GRAVITY = 9.8f;
    
    printf("PI = %.5f\n", PI);
    printf("MAX_SIZE = %d\n", MAX_SIZE);
    printf("最小年龄: %d\n", MIN_AGE);
    printf("重力加速度: %.1f\n", GRAVITY);
    
    return 0;
}
```

## 3. 输入输出

### printf 格式化输出

```c
#include <stdio.h>

int main() {
    int age = 25;
    char name[] = "张三";
    float height = 175.5f;
    
    // 基本格式化
    printf("姓名: %s\n", name);
    printf("年龄: %d\n", age);
    printf("身高: %.2f cm\n", height);
    
    // 宽度和对齐
    printf("姓名: %10s\n", name);      // 右对齐，宽度10
    printf("年龄: %-10d\n", age);      // 左对齐，宽度10
    printf("身高: %10.2f\n", height);  // 宽度10，保留2位小数
    
    return 0;
}
```

**常用格式说明符：**
- `%d` - 十进制整数
- `%o` - 八进制整数
- `%x` - 十六进制整数（小写）
- `%X` - 十六进制整数（大写）
- `%u` - 无符号整数
- `%f` - 浮点数
- `%e` - 科学计数法（小写）
- `%E` - 科学计数法（大写）
- `%g` - 自动选择%f或%e
- `%c` - 单个字符
- `%s` - 字符串
- `%p` - 指针地址

### scanf 格式化输入

```c
#include <stdio.h>

int main() {
    int age;
    char name[50];
    float height;
    
    printf("请输入姓名: ");
    scanf("%s", name);
    
    printf("请输入年龄: ");
    scanf("%d", &age);
    
    printf("请输入身高: ");
    scanf("%f", &height);
    
    printf("\n您输入的信息:\n");
    printf("姓名: %s\n", name);
    printf("年龄: %d\n", age);
    printf("身高: %.2f cm\n", height);
    
    return 0;
}
```

**注意：**
- `scanf`中非字符串变量需要使用`&`取地址符
- 字符串变量本身就是地址，不需要`&`

## 4. 运算符

### 算术运算符

```c
#include <stdio.h>

int main() {
    int a = 10, b = 3;
    
    printf("a = %d, b = %d\n", a, b);
    printf("a + b = %d\n", a + b);   // 加法
    printf("a - b = %d\n", a - b);   // 减法
    printf("a * b = %d\n", a * b);   // 乘法
    printf("a / b = %d\n", a / b);   // 整数除法
    printf("a %% b = %d\n", a % b);  // 取余（注意%%才能打印%）
    
    // 浮点数除法
    float x = 10.0f, y = 3.0f;
    printf("\n%.2f / %.2f = %.2f\n", x, y, x / y);
    
    return 0;
}
```

### 关系运算符

```c
#include <stdio.h>

int main() {
    int a = 10, b = 20;
    
    printf("a = %d, b = %d\n", a, b);
    printf("a == b: %d\n", a == b);  // 等于（0或1）
    printf("a != b: %d\n", a != b);  // 不等于
    printf("a < b: %d\n", a < b);    // 小于
    printf("a > b: %d\n", a > b);    // 大于
    printf("a <= b: %d\n", a <= b);  // 小于等于
    printf("a >= b: %d\n", a >= b);  // 大于等于
    
    return 0;
}
```

### 逻辑运算符

```c
#include <stdio.h>
#include <stdbool.h>

int main() {
    bool x = true, y = false;
    
    printf("x = %d, y = %d\n", x, y);
    printf("x && y: %d\n", x && y);  // 逻辑与
    printf("x || y: %d\n", x || y);  // 逻辑或
    printf("!x: %d\n", !x);          // 逻辑非
    
    return 0;
}
```

### 赋值运算符

```c
#include <stdio.h>

int main() {
    int a = 10;
    
    printf("初始值 a = %d\n", a);
    
    a += 5;  // a = a + 5
    printf("a += 5 后: %d\n", a);
    
    a -= 3;  // a = a - 3
    printf("a -= 3 后: %d\n", a);
    
    a *= 2;  // a = a * 2
    printf("a *= 2 后: %d\n", a);
    
    a /= 4;  // a = a / 4
    printf("a /= 4 后: %d\n", a);
    
    a %= 3;  // a = a % 3
    printf("a %%= 3 后: %d\n", a);
    
    return 0;
}
```

### 自增自减运算符

```c
#include <stdio.h>

int main() {
    int a = 5;
    
    printf("初始值: a = %d\n", a);
    
    // 前缀自增
    printf("++a = %d, 现在 a = %d\n", ++a, a);
    
    // 后缀自增
    printf("a++ = %d, 现在 a = %d\n", a++, a);
    
    // 前缀自减
    printf("--a = %d, 现在 a = %d\n", --a, a);
    
    // 后缀自减
    printf("a-- = %d, 现在 a = %d\n", a--, a);
    
    return 0;
}
```

### 运算符优先级

运算符优先级（从高到低）：
1. `()` - 括号
2. `++`, `--` - 自增自减
3. `*`, `/`, `%` - 乘除取余
4. `+`, `-` - 加减
5. `<`, `<=`, `>`, `>=` - 关系运算符
6. `==`, `!=` - 相等运算符
7. `&&` - 逻辑与
8. `||` - 逻辑或
9. `=` - 赋值

## 5. 类型转换

```c
#include <stdio.h>

int main() {
    // 隐式类型转换
    int a = 5;
    float b = 2.5f;
    float result = a + b;  // int自动转换为float
    printf("%d + %.2f = %.2f\n", a, b, result);
    
    // 显式类型转换（强制类型转换）
    int x = 10;
    int y = 3;
    float quotient = (float)x / y;  // 将x转换为float
    printf("%d / %d = %.2f\n", x, y, quotient);
    
    // 字符与整数转换
    char c = 'A';
    printf("字符 '%c' 的ASCII码: %d\n", c, c);
    printf("ASCII码 65 对应的字符: %c\n", 65);
    
    return 0;
}
```
